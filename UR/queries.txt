//summary of values in gender
SELECT gender.geschlecht,count(*) 
FROM ( 
      SELECT geschlecht 
      FROM `enkelkind` 
      UNION ALL
      SELECT geschlecht 
      FROM `geschwister` 
      UNION ALL
      SELECT geschlecht 
      FROM `geschwisterkind`
      UNION ALL
      SELECT geschlecht 
      FROM `kind` 
      UNION ALL
      SELECT geschlecht 
      FROM `person` 
) gender
GROUP BY geschlecht 
ORDER BY count(*) DESC

FROM religion
